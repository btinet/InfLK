10           
11           // BMI-Rubriken
12           int lowerEnd = 19;  // untere BMI-Grenze
13           int upperEnd = lowerEnd+5;  // obere BMI-Grenze
14           int startAge = 25;  // startWert der ersten Altersgrenze
15           int weightType = 0; // Gewichtskategorie (-1: untergewichtig; 0: normal; 1: übergewichtig)
16           
17           // Prozess-Status (fertig: ja/nein)
18           boolean done = false;
19

21           for(int i = startAge;i < 65;i=i+10) {
22               if(!done) {
23                   if (age < i) {
24                       if (targetBmi < lowerEnd) weightType = -1;
25                       if (targetBmi > upperEnd) weightType = 1;
26                       done = true;
27                   } else {
28                       lowerEnd++;
29                       upperEnd++;
30                   }
31               }
32           }
33   
34           // Für 65-Jährige und ältere
35           if(!done && age >= 65) {
36               if (targetBmi < lowerEnd++) weightType = -1;
37               if (targetBmi > upperEnd++) weightType = 1;
38           }
39   
40           System.out.printf("BMI: %s und Bewertung: %s",targetBmi,weightType);
