1    package aufgaben;

4    import engine.Task;

6    import java.io.IOException;
7    import java.util.Scanner;
8    import java.util.concurrent.ThreadLocalRandom;
9    
10   public class DiceTask extends Task {
11   
12       int tipp;
13       int startGuthaben = 5;
14       int aktuellesGuthaben = -1;
15   
16       Scanner scanner = new Scanner(System.in);
17   
18       public DiceTask() {
19           setTitle("Würfelspiel");
20           setDescription(
21                   "Du hast ein Guthaben von "+startGuthaben+"€. Tippe eine Zahl. Der Computer würfelt dreimal.\n\b" +
22                   "1 Treffer: 1€, 2 Treffer: 2€, 3 Treffer: 5€. Andernfalls verlierst du 1€!"
23           );
24       }
25   
26       public void init () throws IOException {
27           printInfo();
28           getInput();
29       }
30   
31       public void getInput () throws IOException {
32   
33           while (true) {
34               try {
35                   System.out.println();
36                   System.out.print("Dein Tipp: ");
37                   String input = scanner.next(); // Input.console.readLine();
38                   if(input.equals("exit")) {
39                       break;
40                   } else if(Integer.parseInt(input) >= 1 && Integer.parseInt(input) <= 6){
41                       tipp = Integer.parseInt(input);
42                       rollDice();
43                   } else {
44                       throw new NumberFormatException();
45                   }
46   
47               } catch (NumberFormatException exception) {
48                   System.out.println("Zahl zwischen 1 und 6 oder 'exit' eingeben.");
49               } catch (IOException e) {
50                   throw new RuntimeException(e);
51               }
52           }
53           exit();
54       }
55   
56       protected void rollDice () throws IOException {
57   
58           int treffer = 0;
59           if(aktuellesGuthaben == -1) { // Beim ersten Start ist das Guthaben auf -1 gesetzt.
60               aktuellesGuthaben = startGuthaben; // Startguthaben ist 5€.
61           } else if(aktuellesGuthaben == 0) {
62               System.out.println("Du hast kein Guthaben mehr. Bis zum nächsten Mal!");
63               exit();
64           }
65   
66           for (int i = 1; i <= 3; i++) {
67               int currentDiceRoll = ThreadLocalRandom.current().nextInt(1,6 + 1);
68   
69               if(tipp == currentDiceRoll) {
70                   treffer++;
71                   System.out.printf("Wurf %s ergibt: %s Augen. Du triffst!%n",i,currentDiceRoll);
72               } else {
73                   System.out.printf("Wurf %s ergibt: %s Augen. Leider daneben!%n",i,currentDiceRoll);
74               }
75           }
76   
77           switch (treffer) {
78               case 1:
79                   aktuellesGuthaben = aktuellesGuthaben + 1;
80                   System.out.printf("Gut gemacht! Du gewinnst 1€. Du hast nun: %s€.%n",aktuellesGuthaben);
81                   break;
82               case 2:
83                   aktuellesGuthaben = aktuellesGuthaben + 2;
84                   System.out.printf("Gut gemacht! Du gewinnst 2€. Du hast nun: %s€.%n",aktuellesGuthaben);
85                   break;
86               case 3:
87                   aktuellesGuthaben = aktuellesGuthaben + 5;
88                   System.out.printf("Gut gemacht! Du gewinnst 5€. Du hast nun: %s€.%n",aktuellesGuthaben);
89                   break;
90               default:
91                   aktuellesGuthaben--;
92                   System.out.printf("Oh nein! Du verlierst 1€. Du hast nun: %s€.%n",aktuellesGuthaben);
93                   break;
94           }
95   
96       }
